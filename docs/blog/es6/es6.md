## 解构赋值

### 解构赋值的分类

`数组` `对象` `字符串` `boolean` `函数参数` `数值`

```js
let a,b,rest;
[a,b,...rest] = [1,2,3,4,5,6]
// a->1 b->2 rest->[3,4,5,6]

let c,d
{{c,d} = {c:1,d:2}}
// c->1 d->2
```

### 应用场景

```js
// 变量交换
let a = 1;
let b = 2;
[a,b] = [b,a]
// a->2 b->1
```

```js
// 数组操作
let arr = [
  [0,{a:1}],
  [1,{a:2,b:2}],
  [3,{a:3,b:3,c:3}]
]

arr.map(([x,y]) => ({...y,"id":x}))

[
  {id:0,a:1},
  {id:1,a:2,b:2},
  {id:1,a:3,b:3,c:3}
]
```
